# This function calculates and return
# polynomial values of given parameters
function [y] = customPolyVal(p,x)
 y = zeros(1,numel(x));
 n = numel(p);
 for i=n:-1:1
  y = y + p(n+1-i)*x.^(i-1);
 end
end

# Constructing Vendermode matrix
# according to order number
function [D] = genPolyMatrix(x,n)
  D = ones(numel(x),n+1);
  for i=2:(n+1)
    D(:,i) = (x).^(i-1);
  end
end
